// Ensure that the code runs after the Telegram WebApp initializes window.onload = function () { const tg = window.Telegram.WebApp; // Check if the Telegram WebApp API is available if (tg) { const user = tg.initDataUnsafe?.user; if (user) { // Display user data document.getElementById('profile-pic').src = user.photo_url || 'https://via.placeholder.com/100'; document.getElementById('username').textContent = `Username: @${user.username || 'N/A'}`; document.getElementById('user-id').textContent = `User ID: ${user.id}`; document.getElementById('first-name').textContent = `First Name: ${user.first_name || 'N/A'}`; document.getElementById('last-name').textContent = `Last Name: ${user.last_name || 'N/A'}`; } else { alert('Unable to fetch user data.'); } // Match the Telegram theme document.body.style.backgroundColor = tg.themeParams?.bg_color || '#f0f0f0'; document.body.style.color = tg.themeParams?.text_color || '#000'; } else { console.error('Telegram WebApp API is not available.'); } };
